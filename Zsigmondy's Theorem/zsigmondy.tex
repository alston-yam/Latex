    This should serve as a list of notes summarised from \href{https://www.awesomemath.org/wp-pdf-files/math-reflections/mr-2025-02/mr_2_2025_zsigmondys_theorem.pdf}{this handout}.
    
    \begin{definition}[Zsigmondy's Theorem]
        Let $a, b$ be coprime positive integers such that $a \geq b$, and let $n$ be some positive integer greater than 2. Then there exists a prime divisor of $a^n - b^n$ that does not divide $a^k - b^k$ for all $1 \leq k < n$, except the cases when:
        \begin{itemize}
            \item $n=2$ and $a + b$ is a power of 2
            \item $(a, b, n) = (2, 1, 6)$
        \end{itemize}
    \end{definition}
